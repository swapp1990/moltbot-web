**Task 5.3:** Create Simple Workflow
```python
# backend/products/moltbot_test/workflows/greeting.py
from core.langgraph import BaseWorkflow
from core.llm import get_llm

class GreetingWorkflow(BaseWorkflow):
    """Simple workflow that generates a personalized greeting."""

    async def _build_graph(self):
        self.add_node("generate_greeting", handler=self._generate_greeting)
        self.set_entry_point("generate_greeting")
        self.set_end_point("generate_greeting")

    async def _generate_greeting(self, state: dict) -> dict:
        llm = get_llm()
        name = state.get("input", {}).get("name", "Friend")

        prompt = f"Generate a friendly, creative greeting for {name}. Make it warm and personable."
        response = await llm.generate(prompt)

        return {
            **state,
            "output": {
                "greeting": response.content,
                "timestamp": state.get("timestamp")
            }
        }
```

```typescript
// packages/moltbot-test/src/api/client.ts
import axios from 'axios';

const API_BASE = 'http://localhost:8000/api/moltbot_test';

export const createGreetingJob = async (name: string) => {
  const response = await axios.post(`${API_BASE}/jobs`, {
    type: 'greeting',
    input: { name }
  });
  return response.data;
};

export const getJobStatus = async (jobId: string) => {
  const response = await axios.get(`${API_BASE}/jobs/${jobId}`);
  return response.data;
};
```

**Task 6.4:** Create Main Component
```typescript
// packages/moltbot-test/src/components/MoltbotTest.tsx
import React, { useState } from 'react';
import { createGreetingJob, getJobStatus } from '../api/client';

export const MoltbotTest: React.FC = () => {
  const [name, setName] = useState('');
  const [jobId, setJobId] = useState('');
  const [result, setResult] = useState('');
  const [loading, setLoading] = useState(false);

  const handleSubmit = async () => {
    setLoading(true);
    const job = await createGreetingJob(name);
    setJobId(job.id);

    // Poll for result
    const interval = setInterval(async () => {
      const status = await getJobStatus(job.id);
      if (status.status === 'completed') {
        setResult(status.result.greeting);
        setLoading(false);
        clearInterval(interval);
      }
    }, 2000);
  };

  return (
    <div style={{ padding: '20px' }}>
      <h1>Moltbot Test Product</h1>
      <input
        value={name}
        onChange={(e) => setName(e.target.value)}
        placeholder="Enter your name"
      />
      <button onClick={handleSubmit} disabled={loading}>
        {loading ? 'Generating...' : 'Generate Greeting'}
      </button>
      {result && (
        <div style={{ marginTop: '20px', padding: '10px', background: '#f0f0f0' }}>
          <h3>Result:</h3>
          <p>{result}</p>
        </div>
      )}
    </div>
  );
};
```

## 6. Troubleshooting Guide

### Issue: API Key Authentication Failed
**Solution:**
1. Check logs for API errors: `journalctl --user -u moltbot-gateway -f | grep -i "api\|auth"`
2. Verify API key is correct:
   ```bash
   # On droplet
   nano ~/.clawdbot/moltbot.json
   # Look for "apiKey" field under Anthropic config
   ```
3. Test API key manually:
   ```bash
   curl https://api.anthropic.com/v1/messages \
     -H "x-api-key: YOUR_API_KEY" \
     -H "anthropic-version: 2023-06-01" \
     -H "content-type: application/json" \
     -d '{"model":"claude-opus-4-5-20251101","max_tokens":10,"messages":[{"role":"user","content":"Hi"}]}'
   ```
4. If still failing, regenerate API key at https://console.anthropic.com

### Issue: Rate Limit or Credits Exhausted
**Solution:**
1. Check credit balance at https://console.anthropic.com
2. View usage in Moltbot logs
3. If $1000 credits exhausted, add payment method to continue
4. Consider implementing usage limits in Moltbot config

### Issue: Moltbot not responding on WhatsApp
**Solution:**
1. Check gateway status: `systemctl --user status moltbot-gateway`
2. View logs: `journalctl --user -u moltbot-gateway -f`
3. Verify WhatsApp connection in logs (look for "connected")
4. Restart gateway: `systemctl --user restart moltbot-gateway`

### Issue: SSH to MacBook fails
**Solution:**
1. Verify Tailscale is running on both machines
2. Test ping: `ping macbook.tailnet-xyz.ts.net`
3. Test SSH manually: `ssh swapnilsawant@macbook.tailnet-xyz.ts.net`
4. Check SSH keys: `ssh-add -l` (on droplet)

### Issue: Job stuck in "pending" status
**Solution:**
1. Check backend is running: `pgrep -af run_dev.py`
2. View backend logs: `tail -f backend/logs/*.log`
3. Check MongoDB connection
4. Verify worker is polling: look for "Polling for jobs" in logs

### Issue: Frontend can't reach backend
**Solution:**
1. Verify backend running on port 8000
2. Check CORS configuration in backend
3. Test API directly: `curl http://localhost:8000/api/admin/health`
4. Check browser console for errors
